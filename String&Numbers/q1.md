# Question
reversing a string

# Answer
### method 1
```javascript
function reverse(str) {
  new_str = ''
  for(let i=str.length-1; i >= 0; i--) {
    new_str += str[i];
  }
  return new_str
}
```

### method 2
```javascript
let reverse2 = str => str.split('').reverse().join('');
```

### method 3
```javascript
function reverse3(str) {
  str = str.split('');
  let temp = '';
  let len = str.length;
  let half = Math.floor(str.length/2)
  for(let i=0; i < half; i++) {
    temp = str[i];
    str[i] = str[len-i-1];
    str[len-i-1] = temp;
  }
  return str.join('')
}
```
