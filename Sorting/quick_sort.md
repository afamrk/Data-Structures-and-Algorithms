# Quick Sort

```python
def quick_sort(arr):
    qs(arr, 0, len(arr) - 1)
    return arr

def qs(arr, l, r):
    if l >= r:
        return arr
    p = partition(arr, l, r)

    qs(arr, l, p - 1)
    qs(arr, p + 1, r)

def partition(arr, l, r):
    pivot = arr[r]
    i = l - 1
    for j in range(l, r):
        if arr[j] < pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i + 1], arr[r] = arr[r], arr[i + 1]
    return i + 1
```
